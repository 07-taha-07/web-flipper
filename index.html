// Arayüzdeki SİNYAL YAKALA butonu tetikleyicisi function startScanning() { updateTerminal("SİNYAL DİNLENİYOR..."); startNFCScan((result) => { if (result.status === "SUCCESS") { updateTerminal(`SİNYAL KAPILDI! UID: ${result.uid}`); // Burası önemli: Eğer ID boş değilse ekle if (result.uid) { addHistory(result.uid); } } else { updateTerminal(result.message); } }); } function addHistory(uid) { const list = document.getElementById('history'); // Aynı ID zaten varsa tekrar ekleme if (document.getElementById(`uid-${uid}`)) { updateTerminal("SİNYAL ZATEN LİSTEDE."); return; } const item = document.createElement('div'); item.className = 'history-item'; item.id = `uid-${uid}`; // Benzersiz ID atadık item.innerHTML = ` UID: ${uid} `; // Listeye ekle (list'in içini temizlemiyoruz, append yapıyoruz) list.prepend(item); }
